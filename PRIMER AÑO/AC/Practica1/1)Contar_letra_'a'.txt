CONTAR LETRAS 'a'
ORG 1000H
MENSAJE DB "aaaaaaaa"
FIN DB ?
CANT DB ?
ORG 2000H
MOV CL , OFFSET FIN - OFFSET MENSAJE
MOV CH , 'a'
MOV BX, OFFSET MENSAJE
MOV AH, 0
CICLO: MOV AL,[BX]
       CMP AL,CH
       JZ SUMAR
       JMP SEGUIR
SUMAR: INC AH
SEGUIR:INC BX
       DEC CL
       CMP CL,0
       JNZ CICLO
MOV CANT,AH
INT 0
END
